local flyBodyVel, flyBodyGyro

    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    local hrp = char.HumanoidRootPart
    flyBodyVel = Instance.new("BodyVelocity", hrp)
    flyBodyVel.Velocity = Vector3.new(0, 0, 0)
    flyBodyVel.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    flyBodyGyro = Instance.new("BodyGyro", hrp)
    flyBodyGyro.P = 9e4
    flyBodyGyro.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)

    if flyBodyGyro then flyBodyGyro:Destroy() end
end

        flyBodyGyro.CFrame = camCF
    end

local objects = {}
local function createDrawings(player)
    local d = {Line = Drawing.new("Line"), Box = Drawing.new("Square"), Skeleton = {}}
    for i=1, 10 do d.Skeleton[i] = Drawing.new("Line") end
    d.Line.Thickness = 1
    d.Box.Thickness = 1
    objects[player] = d
end

local function updateESP()
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character then
            if not objects[p] then createDrawings(p) end
            local d = objects[p]
            local char = p.Character
            local hrp = char:FindFirstChild("HumanoidRootPart")
            local hum = char:FindFirstChild("Humanoid")
            local head = char:FindFirstChild("Head")
            local teamColor = p.TeamColor.Color or Color3.fromRGB(255, 255, 255)
            if hum and hum.Health > 0 and hrp and head then
                local hrpPos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
                if onScreen then
                    d.Line.Visible = espLinesEnabled
                    d.Line.From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y)
                    d.Line.To = Vector2.new(hrpPos.X, hrpPos.Y)
                    d.Line.Color = teamColor
                    d.Box.Visible = espBoxEnabled
                    local headPos = Camera:WorldToViewportPoint(head.Position + Vector3.new(0, 0.5, 0))
                    local legPos = Camera:WorldToViewportPoint(hrp.Position - Vector3.new(0, 3, 0))
                    d.Box.Size = Vector2.new(2500 / hrpPos.Z, headPos.Y - legPos.Y)
                    d.Box.Position = Vector2.new(hrpPos.X - d.Box.Size.X / 2, hrpPos.Y - d.Box.Size.Y / 2)
                    d.Box.Color = teamColor
                else
                    d.Line.Visible = false
                    d.Box.Visible = false
                end
                if espSkeletonEnabled and onScreen then
                    local torso = char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso")
                    if torso then
                        local p1 = Camera:WorldToViewportPoint(head.Position)
                        local p2 = Camera:WorldToViewportPoint(torso.Position)
                        d.Skeleton[1].Visible = true
                        d.Skeleton[1].From = Vector2.new(p1.X, p1.Y)
                        d.Skeleton[1].To = Vector2.new(p2.X, p2.Y)
                        d.Skeleton[1].Color = teamColor
                    else d.Skeleton[1].Visible = false end
                else
                    for i=1, #d.Skeleton do d.Skeleton[i].Visible = false end
                end
            else
                d.Line.Visible = false
                d.Box.Visible = false
                for i=1, #d.Skeleton do d.Skeleton[i].Visible = false end
            end
        elseif objects[p] then
            objects[p].Line.Visible = false
            objects[p].Box.Visible = false
            for i=1, #objects[p].Skeleton do objects[p].Skeleton[i].Visible = false end
        end
    end
end

-- [[ LÓGICA AIMBOT ]]
local function isVisible(part)
    local char = LocalPlayer.Character
    if not char then return false end
    local params = RaycastParams.new()
    params.FilterDescendantsInstances = {char, Camera}
    params.FilterType = Enum.RaycastFilterType.Exclude
    local result = workspace:Raycast(Camera.CFrame.Position, (part.Position - Camera.CFrame.Position).Unit * 1000, params)
    return result and result.Instance:IsDescendantOf(part.Parent)
end

local function getClosest()
    local target, dist = nil, math.huge
    local center = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character and p.Character:FindFirstChild("Head") then
            local head = p.Character.Head
            if isVisible(head) then
                local pos, onScreen = Camera:WorldToViewportPoint(head.Position)
                if onScreen then
                    local d = (Vector2.new(pos.X, pos.Y) - center).Magnitude
                    if d <= fovCircle.Radius and d < dist then
                        target, dist = p, d
                    end
                end
            end
        end
    end
    return target
end
